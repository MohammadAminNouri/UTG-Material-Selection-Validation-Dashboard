name: Streamlit Validation Check

on:
  push:
    branches:
      - main
      - master
  pull_request:
    branches:
      - main
      - master
  workflow_dispatch:

jobs:
  streamlit-check:
    name: Check UTG Streamlit Dashboard
    runs-on: ubuntu-latest

    steps:
      - name: Checkout repository
        uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v5
        with:
          python-version: "3.11"

      - name: Upgrade pip
        run: |
          python -m pip install --upgrade pip

      - name: Install dependencies
        run: |
          pip install -r requirements.txt

      - name: Check Python syntax
        run: |
          python -m py_compile app.py

      - name: Check source files compile
        run: |
          if [ -d src ]; then python -m compileall src; fi

      - name: Check main Streamlit app exists
        run: |
          test -f app.py

      - name: Run Streamlit smoke test
        run: |
          streamlit run app.py --server.headless true --server.port 8501 &
          sleep 20
          curl --fail http://localhost:8501
